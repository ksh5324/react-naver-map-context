# 지도 옵션 조정하기

지도의 옵션(지도 인터랙션, 관성 드래깅, 타일 fadeIn 효과, 지도 컨트롤, 최소/최대 줌 레벨)을 설정, 해제하는 예제입니다.

지도 초기화 시 지도의 인터랙션과 타일 fadeIn 효과는 기본적으로 활성화되어 있습니다. 지도 옵션에 대한 자세한 내용은 [MapOptions](https://navermaps.github.io/maps.js.ncp/docs/naver.maps.html#.MapOptions)를 참고하세요.

> original docs : https://navermaps.github.io/maps.js.ncp/docs/tutorial-2-map-options.example.html

<LiveEditor>
  {`
import React, { useRef, useState } from "react";
import {Map} from "react-naver-map-sdk";
import NaverMapLoadProvider from "./NaverMapLoadProvider";
import "./global.css";

export default function App() {
  const ref = useRef<naver.maps.Map>();
  
  const [isInteractionEnabled, setIsInteractionEnabled] = useState(true);
  const [isKineticPanEnabled, setIsKineticPanEnabled] = useState(true);
  const [isTileTransitionEnabled, setIsTileTransitionEnabled] = useState(true);
  const [isControlsEnabled, setIsControlsEnabled] = useState(true);
  const [currentZoomRange, setCurrentZoomRange] = useState<"7~21" | "10~21">("10~21");

  const handleInteraction = (e: React.MouseEvent) => {
    e.preventDefault();
    setIsInteractionEnabled(!isInteractionEnabled);

    ref.current?.setOptions({
      draggable: !isInteractionEnabled,
      pinchZoom: !isInteractionEnabled,
      scrollWheel: !isInteractionEnabled,
      keyboardShortcuts: !isInteractionEnabled,
      disableDoubleTapZoom: isInteractionEnabled,
      disableDoubleClickZoom: isInteractionEnabled,
      disableTwoFingerTapZoom: isInteractionEnabled
    });

  };

  const handleKineticPan = (e: React.MouseEvent) => {
    e.preventDefault();
    setIsKineticPanEnabled(!isKineticPanEnabled);
    ref.current?.setOptions("disableKineticPan", isKineticPanEnabled);
  };

  const handleTileTransition = (e: React.MouseEvent) => {
    e.preventDefault();
    setIsTileTransitionEnabled(!isTileTransitionEnabled);
    ref.current?.setOptions("tileTransition", !isTileTransitionEnabled);
  };

  const handleControls = (e: React.MouseEvent) => {
    e.preventDefault();
    setIsControlsEnabled(!isControlsEnabled);
    ref.current?.setOptions({
    scaleControl: !isControlsEnabled,
    logoControl: !isControlsEnabled,
    mapDataControl: !isControlsEnabled,
    zoomControl: !isControlsEnabled,
    mapTypeControl: !isControlsEnabled
    });
  };

  const handleZoomRange = (e: React.MouseEvent) => {
    e.preventDefault();
    const newRange = currentZoomRange === "10~21" ? "7~21" : "10~21";
    setCurrentZoomRange(newRange);
    ref.current?.setOptions({
    minZoom: newRange === "10~21" ? 10 : 7,
    maxZoom: 21
    });
  };
  return (
    <NaverMapLoadProvider>
      <Map
        mapId="map"
        mapOptions={{
          center: { lat: 37.3595704, lng: 127.105399 },
          zoom: 13,
          minZoom: 10,
          zoomControl: true,
          mapTypeControl: true,
        }}
        onZoomChanged={(zoom) => {
          console.log("zoom:" + zoom);
        }}
        onInit={() => {
          console.log(
            "올바른 참조 시점",
            ref.current?.getOptions("minZoom") === 10
          );
        }}
        style={{
          width: "100%",
          height: "350px"
        }}
        ref={ref}
      >
        <div
          className="buttons"
          style={{ position: "absolute", top: 10, right: 10, zIndex: 1000 }}
        >
          <button
            onClick={handleInteraction}
            className={isInteractionEnabled ? "active" : "inactive"}
          >
            지도 인터렉션
          </button>
          <button
            onClick={handleKineticPan}
            className={isKineticPanEnabled ? "active" : "inactive"}
          >
            관성 드래깅
          </button>
          <button
            onClick={handleTileTransition}
            className={isTileTransitionEnabled ? "active" : "inactive"}
          >
            타일 fadeIn 효과
          </button>
          <button
            onClick={handleControls}
            className={isControlsEnabled ? "active" : "inactive"}
          >
            모든 지도 컨트롤
          </button>
          <button onClick={handleZoomRange}>
            {"최소/최대 줌 레벨: " + currentZoomRange}
          </button>
        </div>
      </Map>
    </NaverMapLoadProvider>
  ); 
}

`}

</LiveEditor>
